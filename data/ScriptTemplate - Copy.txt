label(ret_iniConverter)
label(exit)



cave_iniConverter:
PUSHF
PUSH RSI
PUSH RDI
PUSH RCX
mov rcx, E
mov rsi, str1
mov rdi, rbx
rep cmpsb
jne code_iniConverter
mov [rax+08], 0
jmp code_iniConverter

str1:
db 'CARD/MAX_CARDS', 0

code_iniConverter:
  mov ebx,[rax+08]
  jmp exit

exit:
  POP RCX
  POP RDI
  POP RSI
  POPF
  jmp INJECT_iniConverter+08

INJECT_iniConverter+01:
  jmp cave_iniConverter
ret_iniConverter:
registersymbol(INJECT_iniConverter)
